import java.util.Scanner;

public class THIRD {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        
        System.out.print("Enter a number: ");
        int num = sc.nextInt();

        
        int absValue = Math.abs(num);

        
        System.out.println("Absolute value: " + absValue);

        sc.close();
    }
}
